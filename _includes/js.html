Extra JS

Slider: 
function updateSlider(slider) {
    const value = slider.value;
    const percent = value + '%';
    slider.parentElement.style.setProperty('--percent', percent);
    const tooltip = slider.parentElement.querySelector('.tooltip');
    tooltip.textContent = value;
}
Example usage: 
<label class="slider" style="--percent: 53%">
    <input type="range" value="53" min="0" max="100" step="1" oninput="updateSlider(this)">
    <span class="track"></span>
    <span class="thumb"></span>
    <span class="tooltip">53</span>
</label>
Closing CSS-only dialog with ESC: 
document.onkeydown = function (e) {
    if (e.keyCode == 27) {
        var mods = document.querySelectorAll('.dialog-trigger+[type=checkbox]');
        [].forEach.call(mods, function (mod) {
            mod.checked = false;
        });
    }
}
